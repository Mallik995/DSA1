public static Node insertnew(Node head,int K, int P) { 
    if(P==1)
    {
        Node temp=new Node(K);
        temp.next=head;
        head.prev=temp;
        head=temp;
        return head;
    }
    else
    {
        Node tempHead=head; // temporary head
        // go to Pth posiition
        int count=1;
        while(count!=P-1 && tempHead!=null)
        {
            count++;
            tempHead=tempHead.next;
        }

        //Now tempHead will be at P-1th posiition
        Node newNode=new Node(K);
        newNode.next=tempHead.next;
        tempHead.next.prev=newNode;
        tempHead.next=newNode;
        newNode.prev=tempHead;
        return head;
    }
    
    }
